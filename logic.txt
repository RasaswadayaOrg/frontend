function toMatchPercent(score: unknown) {
  const numericScore = Number(score);
  if (!Number.isFinite(numericScore)) return null;
  return Math.max(0, Math.min(100, Math.round(numericScore <= 1 ? numericScore * 100 : numericScore)));
}
